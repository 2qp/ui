## Usage

```tsx
import { RHFTextField } from "@/components/rhf/rhf-text-field";
```

```tsx
<Form {...form}>
  <form onSubmit={form.handleSubmit(onSubmit)}>
    <RHFTextField
      control={form.control}
      name="text"
      label="Name"
      description="something..."
    />
  </form>
</Form>
```
